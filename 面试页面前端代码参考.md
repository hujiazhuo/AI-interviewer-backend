          <text>{{ item.content }}</text>

          </text>
        </view>
      </view>

      <view v-if="thinking" class="row ai" id="m-thinking">
        <view class="bubble ai-bubble thinking">
          <text>正在思考</text>

          <text class="dot">.</text>
        </view>
      </view>
      <view class="bottom-anchor" id="m-bottom"></view>
    </scroll-view>

    <view v-if="awaitChoice" class="decision-wrap">
      <text class="decision-title">阶段性总结完成，你想如何继续？</text>
      <text class="decision-desc">建议：如果你想继续冲刺，可选择“继续问答”；如需进入报告复盘，可选择“结束面试”。系统将在约 5 秒内响应。</text>
      <view class="decision-actions">
        <button
          v-if="decisionOptions.includes('continue')"
          class="decision-btn secondary"

    </view>

<script setup>
import { computed, nextTick, ref } from 'vue'
import { onLoad } from '@dcloudio/uni-app'
import { analyzeInterviewApi, answerInterviewApi, endInterviewApi, startInterviewApi, submitInterviewDecisionApi } from '@/api'

const RESUME_DRAFT_KEY = 'resume_edit_draft_v1'
const jobRole = ref('目标岗位')
const questionMode = ref('balanced')
const randomLevel = ref('high')
const sessionId = ref('')
const currentRound = ref(1)
const totalRounds = ref(10)
const inputText = ref('')
const thinking = ref(false)
const isFinished = ref(false)
const reportLoading = ref(false)
const inputFocus = ref(true)
const scrollToId = ref('m-bottom')
const messages = ref([])
const awaitChoice = ref(false)
const decisionOptions = ref(['end', 'continue'])
const decisionDelayMs = ref(5000)
const decisionLoading = ref(false)
const latestTrace = ref([])
const retrievalTopSource = ref('')
const routeTimeline = ref([])

const modeText = computed(() => {
  const map = {
    balanced: '均衡出题',
    'project-lite': '减少项目追问',
    'project-deep': '项目深挖',
  }
  return map[questionMode.value] || '均衡出题'
})

const randomText = computed(() => {
  const map = {
    low: '低',
    medium: '中',
    high: '高',
  }
  return map[randomLevel.value] || '高'
})

onLoad((options) => {
  jobRole.value = decodeURIComponent(options?.job_role || '目标岗位')
  questionMode.value = decodeURIComponent(options?.mode || 'balanced')
  randomLevel.value = decodeURIComponent(options?.rand || 'high')
  startInterview()
})

async function startInterview() {
  thinking.value = true
  try {
    const data = await startInterviewApi({
      job_role: jobRole.value,
      resume: getResumePayload(),
      question_policy: buildQuestionPolicy(),
      knowledge_scope: getKnowledgeScope(),
      asked_questions: getAskedQuestions(),
    })
    sessionId.value = data?.sessionId || data?.session_id || ''
    currentRound.value = Number(data?.currentRound || data?.round || 1)
    totalRounds.value = Number(data?.totalRounds || data?.total_rounds || 10)

    const firstQuestion =
      data?.question ||
      data?.firstQuestion ||
      `你好，我是你的 AI 面试官。我们现在开始 ${jobRole.value} 的模拟面试。请先做一个 1 分钟自我介绍。`

    applyMeta(data)
    pushMessage('ai', firstQuestion, data?.routeHint, currentRound.value)
  } catch (error) {
    pushMessage('ai', `你好，我们开始 ${jobRole.value} 面试。请先做一个简短自我介绍。`)
    uni.showToast({ title: '已进入本地模拟模式', icon: 'none' })
  } finally {
    thinking.value = false
    refocusInput()
  }
}

async function sendAnswer() {
  if (awaitChoice.value) {
    uni.showToast({ title: '请先选择结束或继续问答', icon: 'none' })
    return
  }

  if (isFinished.value) {
    uni.showToast({ title: '本轮面试已结束', icon: 'none' })
    return
  }

  const answer = inputText.value.trim()
  if (!answer || thinking.value) return

  pushMessage('user', answer)
  inputText.value = ''
  thinking.value = true

  try {
    const data = await answerInterviewApi({
      sessionId: sessionId.value,
      job_role: jobRole.value,
      answer,
      round: currentRound.value,
      question_policy: buildQuestionPolicy(),
      knowledge_scope: getKnowledgeScope(),
      asked_questions: getAskedQuestions(),
    })

    const answeredRound = Number(data?.answeredRound || data?.currentRound || data?.round || currentRound.value)
    const nextQuestionRound = Number(data?.nextQuestionRound || answeredRound + 1)

    totalRounds.value = Number(data?.totalRounds || data?.total_rounds || totalRounds.value)

    const nextQuestion =
      data?.nextQuestion ||
      data?.question ||
      data?.reply ||
      '收到你的回答。请进一步说明你在该项目中的关键决策和结果。'

    const optionsFromApi = readDecisionOptions(data)
    const phase = String(data?.phase || '').toLowerCase()
    const status = String(data?.status || '').toLowerCase()
    const checkpointFlag =
      isTrue(data?.mustChoose) ||
      phase === 'checkpoint' ||
      status === 'checkpoint'

    const atChoiceNode =
      isTrue(data?.awaitChoice) ||
      isTrue(data?.await_choice) ||
      checkpointFlag ||
      optionsFromApi.length > 0 ||
      looksLikeDecisionPrompt(nextQuestion)

    if (atChoiceNode) {
      isFinished.value = false
      reportLoading.value = false
      currentRound.value = Number.isFinite(answeredRound) ? Math.max(1, answeredRound) : currentRound.value
      applyMeta(data)
      pushMessage('ai', nextQuestion, data?.routeHint, currentRound.value)

      awaitChoice.value = true
      decisionOptions.value = optionsFromApi.length
        ? optionsFromApi
        : ['end', 'continue']
      decisionDelayMs.value = Number(data?.decisionDelayMs || data?.decision_delay_ms || 5000)
      refocusInput()
      return
    }

    currentRound.value = Number.isFinite(nextQuestionRound)
      ? Math.max(1, nextQuestionRound)
      : Math.max(1, currentRound.value + 1)

    applyMeta(data)
    pushMessage('ai', nextQuestion, data?.routeHint, currentRound.value)

    if (!atChoiceNode && (isTrue(data?.finished) || isTrue(data?.is_finished))) {
      isFinished.value = true
      pushMessage('ai', '本轮面试已完成，正在生成报告...')
      await generateReportAndJump()
    }
  } catch (error) {
    const msg = String(error?.message || '')
    if (msg.includes('4040') || msg.includes('会话不存在')) {
      uni.showToast({ title: '会话不存在，请重新开始', icon: 'none' })
      isFinished.value = true
      return
    }
    if (msg.includes('4090') || msg.includes('会话已结束')) {
      uni.showToast({ title: '会话已结束', icon: 'none' })
      isFinished.value = true
      return
    }
    pushMessage('ai', '网络波动，已切换本地追问：请举例说明一次你解决复杂问题的过程。')
  } finally {
    thinking.value = false
    scrollBottom()
    if (!isFinished.value) {
      refocusInput()
    }
  }
}

async function submitDecision(decision) {
  if (decisionLoading.value || !sessionId.value) return
  decisionLoading.value = true
  awaitChoice.value = false
  thinking.value = true

  const label = decision === 'continue' ? '继续问答' : '结束面试'
  pushMessage('user', `我选择：${label}`)

  try {
    const data = await submitInterviewDecisionApi({
      sessionId: sessionId.value,
      job_role: jobRole.value,
      decision,
      question_policy: buildQuestionPolicy(),
      knowledge_scope: getKnowledgeScope(),
      asked_questions: getAskedQuestions(),
    })

    applyMeta(data)

    const wait = Math.max(800, Math.min(8000, Number(data?.decisionDelayMs || decisionDelayMs.value || 5000)))
    await sleep(wait)

    const apiDecision = String(data?.decision || '').toLowerCase()
    if (decision === 'end' || apiDecision === 'end') {
      pushMessage('ai', data?.question || data?.message || '好的，本次面试到此结束，我将为你生成评估报告。')
      isFinished.value = true
      await generateReportAndJump()
      return
    }

    totalRounds.value = Number(data?.totalRounds || data?.total_rounds || totalRounds.value)
    const nextRound = Number(data?.nextQuestionRound || data?.currentRound || data?.round || currentRound.value + 1)
    currentRound.value = Number.isFinite(nextRound) ? Math.max(1, nextRound) : currentRound.value + 1

    const nextQuestion = data?.nextQuestion || data?.question || data?.reply || '继续：请回答下一个问题。'
    pushMessage('ai', nextQuestion, data?.routeHint, currentRound.value)

    if (data?.awaitChoice === true) {
      awaitChoice.value = true
      decisionOptions.value = Array.isArray(data?.decisionOptions) && data.decisionOptions.length
        ? data.decisionOptions
        : ['end', 'continue']
      decisionDelayMs.value = Number(data?.decisionDelayMs || decisionDelayMs.value || 5000)
    }
  } catch (error) {
    uni.showToast({ title: error.message || '提交选择失败', icon: 'none' })
    awaitChoice.value = true
  } finally {
    thinking.value = false
    decisionLoading.value = false
    scrollBottom()
    if (!isFinished.value && !awaitChoice.value) {
      refocusInput()
    }
  }
}

function refocusInput() {
  if (isFinished.value) return
  inputFocus.value = false
  nextTick(() => {
    setTimeout(() => {
      inputFocus.value = true
    }, 0)
  })
}

function applyMeta(data) {
  const trace = Array.isArray(data?.trace) ? data.trace.filter(Boolean) : []
  latestTrace.value = trace
  retrievalTopSource.value = data?.retrieval?.topSource || data?.retrieval?.rerankSource || ''
}

function upsertRouteTimeline(round, routeHint) {
  if (!routeHint) return
  const r = Number(round)
  if (!Number.isFinite(r)) return
  const idx = routeTimeline.value.findIndex((item) => item.round === r)
  if (idx >= 0) {
    routeTimeline.value[idx].routeHint = routeHint
    return
  }
  routeTimeline.value.push({ round: r, routeHint })
  routeTimeline.value.sort((a, b) => a.round - b.round)
}

function getResumePayload() {
  const draft = uni.getStorageSync(RESUME_DRAFT_KEY)
  if (!draft || typeof draft !== 'object') return {}
  return {
    name: draft?.name || '',
    workYears: draft?.workYears || '',
    targetJobIndustry: draft?.targetJobIndustry || '',
    techStack: draft?.techStack || '',
    mastery: draft?.mastery || '',
    directions: Array.isArray(draft?.directions) ? draft.directions : [],
    projects: Array.isArray(draft?.projects) ? draft.projects : [],
  }
}

function getKnowledgeScope() {
  const role = String(jobRole.value || '').toLowerCase()
  if (role.includes('java')) return 'java'
  if (role.includes('网络')) return 'network'
  if (role.includes('大模型') || role.includes('llm')) return 'llm'
  return 'general'
}

function getAskedQuestions() {
  return messages.value
    .filter((m) => m.role === 'ai' && m.content)
    .map((m) => String(m.content).trim())
    .filter(Boolean)
    .slice(-12)
}

function buildQuestionPolicy() {
  const mode = ['balanced', 'project-lite', 'project-deep'].includes(questionMode.value)
    ? questionMode.value
    : 'balanced'
  const randomness = ['low', 'medium', 'high'].includes(randomLevel.value)
    ? randomLevel.value
    : 'high'

  const maxProjectFollowups = mode === 'project-lite' ? 0 : mode === 'project-deep' ? 4 : 2
  return {
    mode,
    randomness,
    avoid_repeat: true,
    diversify_topics: true,
    max_project_followups: maxProjectFollowups,
    avoid_project_chain: mode === 'project-lite',
    project_reference_once: mode === 'project-lite',
  }
}

function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms))
}

function isTrue(val) {
  if (val === true) return true
  const t = String(val || '').toLowerCase()
  return t === 'true' || t === '1' || t === 'yes'
}

function readDecisionOptions(data) {
  const arr = Array.isArray(data?.decisionOptions)
    ? data.decisionOptions
    : Array.isArray(data?.decision_options)
      ? data.decision_options
      : []
  return arr.map((x) => String(x).toLowerCase()).filter(Boolean)
}

function looksLikeDecisionPrompt(text) {
  const s = String(text || '')
  if (!s) return false
  return /选择|继续问答|结束面试|continue|end/i.test(s)
}

async function generateReportAndJump() {
  if (!sessionId.value) {
    uni.redirectTo({ url: '/pages/interview/records' })
    return
  }
  reportLoading.value = true
  try {
    await endInterviewApi({ sessionId: sessionId.value, job_role: jobRole.value })
    const report = await analyzeInterviewApi({
      sessionId: sessionId.value,
      job_role: jobRole.value,
    })
    const id = report?.id || report?.reportId || report?.report_id || sessionId.value
    uni.redirectTo({ url: `/pages/interview/report?id=${encodeURIComponent(id)}` })
  } catch (error) {
    uni.showToast({ title: error.message || '生成报告失败', icon: 'none' })
    uni.redirectTo({ url: '/pages/interview/records' })
  } finally {
    reportLoading.value = false
  }
}

function pushMessage(role, content, routeHint = '', routeRound = 0) {
  messages.value.push({
    id: Date.now() + Math.random(),
    role,
    content,
    routeHint,
  })
  if (role === 'ai' && routeHint) {
    upsertRouteTimeline(routeRound, routeHint)
  }
  scrollBottom()
}

function scrollBottom() {
  nextTick(() => {
    scrollToId.value = 'm-bottom'
  })
}

function confirmEndInterview() {
  uni.showModal({
    title: '结束面试',
    content: '确认结束本次模拟面试吗？',
    confirmText: '结束',
    success: async (res) => {
      if (!res.confirm) return
      isFinished.value = true
      await generateReportAndJump()
    },
  })
}
</script>
